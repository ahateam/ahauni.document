# 测试服

# 端口号配置，一般情况下在 xhj.cn.start 包下的 JITIVerticle.java 中配置

// 集体经济项目端口号为8080
public int port() {
	return 8080;
}

# 接口开放，将需要开放出去的接口类添加 JITIVerticle.java 中的 init 函数中

protected void init() throws Exception {
    							//接口类			  //接口调用名
	initCtrl(ctrlMap, Singleton.ins(ORGController.class, "org"));
}
